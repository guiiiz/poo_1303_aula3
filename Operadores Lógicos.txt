#include <iostream>
#include <string>
using namespace std;

int main() {
    // Declarando as variáveis x e y, e atribuindo valores inteiros para elas
    int x = 10;
    int y = 20;

    // Verificando se x é maior que y e exibindo a mensagem na tela
    if (x > y) {
        cout << "x é maior que y" << endl;
    }

    // Verificando se x é menor que y e exibindo a mensagem na tela
    if (x < y) {
        cout << "x é menor que y" << endl;
    }

    // Verificando se x é maior ou igual a y e exibindo a mensagem na tela
    if (x >= y) {
        cout << "x é maior ou igual a y" << endl;
    }

    // Verificando se x é menor ou igual a y e exibindo a mensagem na tela
    if (x <= y) {
        cout << "x é menor ou igual a y" << endl;
    }

    // Verificando se x é igual a y e exibindo a mensagem na tela
    if (x == y) {
        cout << "x é igual a y" << endl;
    }

    // Verificando se x é diferente de y e exibindo a mensagem na tela
    if (x != y) {
        cout << "x é diferente de y" << endl;
    }

    return 0;
}
